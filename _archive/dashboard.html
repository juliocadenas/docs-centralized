# Instalar: pip install openai
from openai import OpenAI

client = OpenAI(
    base_url="http://100.105.27.27:9000/v1",
    api_key="not-needed"
)

# Chat con LLM
response = client.chat.completions.create(
    model="llama3.1",
    messages=[{"role": "user", "content": "Hola, explica IA en 3 líneas"}]
)
print(response.choices[0].message.content)

# Chat con LLM
curl -X POST http://100.105.27.27:9000/v1/chat/completions \
  -H "Content-Type: application/json" \
  -d '{
    "model": "llama3.1",
    "messages": [{"role": "user", "content": "Hola"}]
  }'

# Generar imagen
curl -X POST http://100.105.27.27:9000/v1/images/generations \
  -H "Content-Type: application/json" \
  -d '{"prompt": "a cyberpunk cat", "model": "sdxl"}'
